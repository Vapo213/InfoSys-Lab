CREATE TRIGGER FRMEIT00.T_DOZ_TOTAL AFTER  INSERT  ON FRMEIT00.DOZENT  REFERENCING  NEW AS newDOZ  FOR EACH ROW  MODE DB2SQL 
WHEN  ( newDOZ.DOZENT_ID NOT IN  (
( SELECT LEHRBEAUFTRAGTER.DOZENT_ID
   FROM FRMEIT00.LEHRBEAUFTRAGTER AS LEHRBEAUFTRAGTER ) 
UNION
( SELECT PROFESSOR.DOZENT_ID
   FROM FRMEIT00.PROFESSOR AS PROFESSOR )
))
BEGIN ATOMIC
SIGNAL SQLSTATE '99003' ('Dozent ist nicht total') ;
END
